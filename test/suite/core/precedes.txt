#!/usr/local/bin/langnd

import ["read", "evaluate", "get"] from core

$testing = $evaluate($read("testing.txt", null))
$returns = $get($testing, "returns")
$throws = $get($testing, "throws")

$returns(<import "precedes" from core $precedes(null, null)>, false)
$returns(<import "precedes" from core $precedes(null, true)>, true)
$returns(<import "precedes" from core $precedes(null, 100)>, true)
$returns(<import "precedes" from core $precedes(null, "text")>, true)
$returns(<import "precedes" from core $precedes(null, [1, 2, 3])>, true)
$returns(<import "precedes" from core $precedes(null, {"a": 1, "b": 2, "c": 3})>, true)

$returns(<import "precedes" from core $precedes(false, true)>, true)
$returns(<import "precedes" from core $precedes(true, false)>, false)

$returns(<import "precedes" from core $precedes(true, null)>, false)

$returns(<import "precedes" from core $precedes(true, {"a": 1, "b": 2, "c": 3})>, true)

$returns(<import "precedes" from core $precedes(-32767.999999, -2147483647)>, false)
$returns(<import "precedes" from core $precedes(100, null)>, false)
