#!/usr/local/bin/langnd

import ["read", "evaluate", "get"] from core

$testing = $evaluate($read("testing.txt", null))
$returns = $get($testing, "returns")
$throws = $get($testing, "throws")

$returns(<import "evaluate" from core $evaluate("null")>, null)
$returns(<import "evaluate" from core $evaluate("false")>, false)
$returns(<import "evaluate" from core $evaluate("0")>, 0)
$returns(<import "evaluate" from core $evaluate("\"\"")>, "")
$returns(<import "evaluate" from core $evaluate("[]")>, [])
$returns(<import "evaluate" from core $evaluate("{}")>, {})
$returns(<import "evaluate" from core $evaluate("<>")>, <>)

$returns(<import "evaluate" from core $evaluate("import \"add\" from core $add(100, 200)")>, 300)

$throws(<import "evaluate" from core $evaluate("...")>, {"hint": "...", "message": "failed to lex code"})
$throws(<import "evaluate" from core $evaluate("[")>, {"hint": "[", "message": "failed to parse code"})
$throws(<import "evaluate" from core $evaluate("throw \"error\"")>, {"hint": "error", "message": "failed to execute code"})

$throws(<import "evaluate" from core $evaluate("return \"lost\"")>, {"hint": "lost return", "message": "failed to execute code"})
$throws(<import "evaluate" from core $evaluate("break \"lost\"")>, {"hint": "lost break", "message": "failed to execute code"})
$throws(<import "evaluate" from core $evaluate("continue \"lost\"")>, {"hint": "lost continue", "message": "failed to execute code"})
