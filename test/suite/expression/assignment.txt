#!/usr/local/bin/langnd

import ["read", "evaluate", "get"] from core

$testing = $evaluate($read("testing.txt", null))
$returns = $get($testing, "returns")
$lexFails = $get($testing, "lexFails")

$returns(<$a=null>, null)
$returns(<$a=true>, true)
$returns(<$a=100>, 100)
$returns(<$a="text">, "text")
$returns(<$a=[1, 2, 3]>, [1, 2, 3])
$returns(<$a={"a": 1, "b": 2, "c": 3}>, {"a": 1, "b": 2, "c": 3})
$returns(<$a=<import "multiply" from core $number=argument return $multiply($number, $number)>>, <import "multiply" from core $number=argument return $multiply($number, $number)>)
$returns(<$A="val">, "val")
$returns(<$0="val">, "val")
$returns(<$_="val">, "val")
$returns(<$""="val">, "val")
$returns(<$"#$\\/()[]{}<>"="val">, "val")
$returns(<$"\t\r\n"="val">, "val")
$returns(<$"\a001\a002\a003"="val">, "val")
$returns(<$Text_100="val">, "val")
$returns(<$"(long) & [special] --> ID"="val">, "val")
$returns(<$a=null $a>, null)
$returns(<$a=true $a>, true)
$returns(<$a=100 $a>, 100)
$returns(<$a="text" $a>, "text")
$returns(<$a=[1, 2, 3] $a>, [1, 2, 3])
$returns(<$a={"a": 1, "b": 2, "c": 3} $a>, {"a": 1, "b": 2, "c": 3})
$returns(<$a=<import "multiply" from core $number=argument return $multiply($number, $number)> $a>, <import "multiply" from core $number=argument return $multiply($number, $number)>)
$returns(<$a=100 $a=null $a>, null)
$returns(<$a=100 $a=200 $a=300 $a>, 300)
$returns(<$a=100 $b=$a $b>, 100)
$returns(<$a=100 $a=$a $a>, 100)
$returns(<import "add" from core $a=100 $b=200 $add($a, $b)>, 300)
$returns(<import "merge" from core $a="this" $b=" & that" $merge($a, $b)>, "this & that")
$returns(<$"long id"="found" $"long id">, "found")
$returns(<$""="found" $"">, "found")
$returns(<$"id"="found" $"id">, "found")
$returns(<$id="found" $"id">, "found")
$returns(<$"id"="found" $id>, "found")
$returns(<$copy=$new="found" $copy>, "found")
